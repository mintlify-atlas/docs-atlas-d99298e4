## Introduction

The Select component is built on top of Radix UI's Select primitive and provides a fully accessible dropdown for selecting values from a list of options.

## Installation

The Select component is available as part of the `@poketto/ui` package:

```tsx
import {
  Select,
  SelectContent,
  SelectItem,
  SelectTrigger,
  SelectValue,
} from "@poketto/ui";
```

## Basic Usage

```tsx
import {
  Select,
  SelectContent,
  SelectItem,
  SelectTrigger,
  SelectValue,
} from "@poketto/ui";

export default function Example() {
  return (
    <Select>
      <SelectTrigger className="w-[180px]">
        <SelectValue placeholder="Select a fruit" />
      </SelectTrigger>
      <SelectContent>
        <SelectItem value="apple">Apple</SelectItem>
        <SelectItem value="banana">Banana</SelectItem>
        <SelectItem value="orange">Orange</SelectItem>
      </SelectContent>
    </Select>
  );
}
```

### Controlled Select

```tsx
import {
  Select,
  SelectContent,
  SelectItem,
  SelectTrigger,
  SelectValue,
} from "@poketto/ui";
import { useState } from "react";

export default function ControlledSelect() {
  const [value, setValue] = useState("");

  return (
    <div className="space-y-2">
      <Select value={value} onValueChange={setValue}>
        <SelectTrigger className="w-[200px]">
          <SelectValue placeholder="Select a color" />
        </SelectTrigger>
        <SelectContent>
          <SelectItem value="red">Red</SelectItem>
          <SelectItem value="green">Green</SelectItem>
          <SelectItem value="blue">Blue</SelectItem>
        </SelectContent>
      </Select>
      {value && (
        <p className="text-sm text-muted-foreground">
          Selected: {value}
        </p>
      )}
    </div>
  );
}
```

## Props

### Select (Root)

<ParamField path="value" type="string">
  The controlled value of the select.
</ParamField>

<ParamField path="defaultValue" type="string">
  The default value for uncontrolled select.
</ParamField>

<ParamField path="onValueChange" type="(value: string) => void">
  Event handler called when the value changes.
</ParamField>

<ParamField path="disabled" type="boolean">
  When `true`, disables the select.
</ParamField>

### SelectTrigger

<ParamField path="className" type="string">
  Additional CSS classes to apply to the trigger.
</ParamField>

### SelectContent

<ParamField path="position" type="string" default="popper">
  The positioning mode for the content.
  
  Options: `"popper"` | `"item-aligned"`
</ParamField>

<ParamField path="className" type="string">
  Additional CSS classes to apply to the content.
</ParamField>

### SelectItem

<ParamField path="value" type="string" required>
  The value of the select item.
</ParamField>

<ParamField path="disabled" type="boolean">
  When `true`, disables the select item.
</ParamField>

<ParamField path="className" type="string">
  Additional CSS classes to apply to the item.
</ParamField>

## Form Integration

The Select component integrates seamlessly with React Hook Form:

```tsx
import {
  Select,
  SelectContent,
  SelectItem,
  SelectTrigger,
  SelectValue,
} from "@poketto/ui";
import { Button } from "@poketto/ui";
import { useForm } from "react-hook-form";
import {
  Form,
  FormControl,
  FormDescription,
  FormField,
  FormItem,
  FormLabel,
  FormMessage,
} from "@poketto/ui";

interface FormData {
  role: string;
}

export default function SelectForm() {
  const form = useForm<FormData>({
    defaultValues: {
      role: "",
    },
  });

  const onSubmit = (data: FormData) => {
    console.log(data);
  };

  return (
    <Form {...form}>
      <form onSubmit={form.handleSubmit(onSubmit)} className="space-y-4">
        <FormField
          control={form.control}
          name="role"
          render={({ field }) => (
            <FormItem>
              <FormLabel>Role</FormLabel>
              <Select onValueChange={field.onChange} defaultValue={field.value}>
                <FormControl>
                  <SelectTrigger>
                    <SelectValue placeholder="Select a role" />
                  </SelectTrigger>
                </FormControl>
                <SelectContent>
                  <SelectItem value="admin">Admin</SelectItem>
                  <SelectItem value="user">User</SelectItem>
                  <SelectItem value="guest">Guest</SelectItem>
                </SelectContent>
              </Select>
              <FormDescription>
                Choose your role in the organization.
              </FormDescription>
              <FormMessage />
            </FormItem>
          )}
        />
        <Button type="submit">Submit</Button>
      </form>
    </Form>
  );
}
```
