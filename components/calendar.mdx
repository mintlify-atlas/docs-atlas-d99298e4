## Introduction

The Calendar component is built on top of react-day-picker and provides a fully accessible and customizable calendar for date selection. It supports single date, multiple dates, and date range selection modes.

## Installation

The Calendar component is available as part of the `@poketto/ui` package:

```tsx
import { Calendar } from "@poketto/ui";
```

The component uses `react-day-picker` which should be included in the package dependencies.

## Basic Usage

```tsx
import { Calendar } from "@poketto/ui";
import { useState } from "react";

export default function Example() {
  const [date, setDate] = useState<Date | undefined>(new Date());

  return (
    <Calendar
      mode="single"
      selected={date}
      onSelect={setDate}
      className="rounded-md border"
    />
  );
}
```

## Examples

### Single Date Selection

```tsx
import { Calendar } from "@poketto/ui";
import { useState } from "react";

export default function SingleDateCalendar() {
  const [date, setDate] = useState<Date | undefined>();

  return (
    <div className="space-y-4">
      <Calendar
        mode="single"
        selected={date}
        onSelect={setDate}
        className="rounded-md border"
      />
      {date && (
        <p className="text-sm text-muted-foreground">
          Selected: {date.toLocaleDateString()}
        </p>
      )}
    </div>
  );
}
```

### Multiple Date Selection

```tsx
import { Calendar } from "@poketto/ui";
import { useState } from "react";

export default function MultipleDateCalendar() {
  const [dates, setDates] = useState<Date[] | undefined>();

  return (
    <div className="space-y-4">
      <Calendar
        mode="multiple"
        selected={dates}
        onSelect={setDates}
        className="rounded-md border"
      />
      {dates && dates.length > 0 && (
        <div className="text-sm text-muted-foreground">
          <p>Selected {dates.length} date(s):</p>
          <ul className="list-disc list-inside">
            {dates.map((date, i) => (
              <li key={i}>{date.toLocaleDateString()}</li>
            ))}
          </ul>
        </div>
      )}
    </div>
  );
}
```

### Date Range Selection

```tsx
import { Calendar } from "@poketto/ui";
import { useState } from "react";
import type { DateRange } from "react-day-picker";

export default function RangeDateCalendar() {
  const [range, setRange] = useState<DateRange | undefined>();

  return (
    <div className="space-y-4">
      <Calendar
        mode="range"
        selected={range}
        onSelect={setRange}
        className="rounded-md border"
        numberOfMonths={2}
      />
      {range?.from && (
        <div className="text-sm text-muted-foreground">
          {range.to ? (
            <p>
              {range.from.toLocaleDateString()} - {range.to.toLocaleDateString()}
            </p>
          ) : (
            <p>{range.from.toLocaleDateString()}</p>
          )}
        </div>
      )}
    </div>
  );
}
```

### Disabled Dates

```tsx
import { Calendar } from "@poketto/ui";
import { useState } from "react";

export default function DisabledDatesCalendar() {
  const [date, setDate] = useState<Date | undefined>();

  // Disable weekends
  const disabledDays = [
    { dayOfWeek: [0, 6] }, // Sunday and Saturday
  ];

  return (
    <Calendar
      mode="single"
      selected={date}
      onSelect={setDate}
      disabled={disabledDays}
      className="rounded-md border"
    />
  );
}
```

### With Min and Max Date

```tsx
import { Calendar } from "@poketto/ui";
import { useState } from "react";

export default function MinMaxCalendar() {
  const [date, setDate] = useState<Date | undefined>();
  const today = new Date();
  const oneMonthFromNow = new Date(today.getFullYear(), today.getMonth() + 1, today.getDate());

  return (
    <Calendar
      mode="single"
      selected={date}
      onSelect={setDate}
      disabled={{ before: today, after: oneMonthFromNow }}
      className="rounded-md border"
    />
  );
}
```

## Props

<ParamField path="mode" type="'single' | 'multiple' | 'range'" required>
  The selection mode of the calendar.
</ParamField>

<ParamField path="selected" type="Date | Date[] | DateRange | undefined">
  The selected date(s). Type depends on the mode:
  - `single`: `Date | undefined`
  - `multiple`: `Date[] | undefined`
  - `range`: `DateRange | undefined`
</ParamField>

<ParamField path="onSelect" type="(date: Date | Date[] | DateRange | undefined) => void">
  Called when a date is selected. The parameter type depends on the mode.
</ParamField>

<ParamField path="disabled" type="Matcher | Matcher[]">
  Dates that should be disabled. Can be:
  - A date object
  - A range `{ from: Date, to: Date }`
  - A function `(date: Date) => boolean`
  - An object with `before`, `after`, or `dayOfWeek` properties
</ParamField>

<ParamField path="defaultMonth" type="Date">
  The month to display by default.
</ParamField>

<ParamField path="numberOfMonths" type="number" default="1">
  The number of months to display.
</ParamField>

<ParamField path="showOutsideDays" type="boolean" default="true">
  Whether to show days from adjacent months.
</ParamField>

<ParamField path="className" type="string">
  Additional CSS classes to apply to the calendar.
</ParamField>

<ParamField path="classNames" type="object">
  Custom class names for internal calendar elements. See [react-day-picker documentation](https://react-day-picker.js.org/api/types/ClassNames) for details.
</ParamField>

The Calendar component also accepts all props from [react-day-picker's DayPicker](https://react-day-picker.js.org/api/interfaces/DayPickerBase).

## Form Integration

The Calendar is commonly used with a Popover to create a date picker within forms:

```tsx
import { zodResolver } from "@hookform/resolvers/zod";
import { format } from "date-fns";
import { CalendarIcon } from "lucide-react";
import { useForm } from "react-hook-form";
import * as z from "zod";
import { Button } from "@poketto/ui";
import { Calendar } from "@poketto/ui";
import {
  Form,
  FormControl,
  FormDescription,
  FormField,
  FormItem,
  FormLabel,
  FormMessage,
} from "@poketto/ui";
import {
  Popover,
  PopoverContent,
  PopoverTrigger,
} from "@poketto/ui";
import { cn } from "@poketto/ui/lib/utils";

const formSchema = z.object({
  dob: z.date({
    required_error: "A date of birth is required.",
  }),
});

export default function CalendarForm() {
  const form = useForm<z.infer<typeof formSchema>>({
    resolver: zodResolver(formSchema),
  });

  function onSubmit(values: z.infer<typeof formSchema>) {
    console.log(values);
  }

  return (
    <Form {...form}>
      <form onSubmit={form.handleSubmit(onSubmit)} className="space-y-4">
        <FormField
          control={form.control}
          name="dob"
          render={({ field }) => (
            <FormItem className="flex flex-col">
              <FormLabel>Date of birth</FormLabel>
              <Popover>
                <PopoverTrigger asChild>
                  <FormControl>
                    <Button
                      variant="outline"
                      className={cn(
                        "w-[240px] pl-3 text-left font-normal",
                        !field.value && "text-muted-foreground"
                      )}
                    >
                      {field.value ? (
                        format(field.value, "PPP")
                      ) : (
                        <span>Pick a date</span>
                      )}
                      <CalendarIcon className="ml-auto h-4 w-4 opacity-50" />
                    </Button>
                  </FormControl>
                </PopoverTrigger>
                <PopoverContent className="w-auto p-0" align="start">
                  <Calendar
                    mode="single"
                    selected={field.value}
                    onSelect={field.onChange}
                    disabled={(date) =>
                      date > new Date() || date < new Date("1900-01-01")
                    }
                    initialFocus
                  />
                </PopoverContent>
              </Popover>
              <FormDescription>
                Your date of birth is used to calculate your age.
              </FormDescription>
              <FormMessage />
            </FormItem>
          )}
        />
        <Button type="submit">Submit</Button>
      </form>
    </Form>
  );
}
```
