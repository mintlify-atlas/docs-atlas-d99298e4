## Introduction

The Checkbox component is built on top of Radix UI's Checkbox primitive and provides a fully accessible checkbox with a check indicator.

## Installation

The Checkbox component is available as part of the `@poketto/ui` package:

```tsx
import { Checkbox } from "@poketto/ui";
```

## Basic Usage

```tsx
import { Checkbox } from "@poketto/ui";

export default function Example() {
  return <Checkbox />;
}
```

### Controlled Checkbox

```tsx
import { Checkbox } from "@poketto/ui";
import { Label } from "@poketto/ui";
import { useState } from "react";

export default function ControlledCheckbox() {
  const [checked, setChecked] = useState(false);

  return (
    <div className="space-y-2">
      <div className="flex items-center space-x-2">
        <Checkbox 
          id="controlled" 
          checked={checked}
          onCheckedChange={(checked) => setChecked(checked as boolean)}
        />
        <Label htmlFor="controlled">Subscribe to newsletter</Label>
      </div>
      <p className="text-sm text-muted-foreground">
        Status: {checked ? "Subscribed" : "Not subscribed"}
      </p>
    </div>
  );
}
```

### Indeterminate State

```tsx
import { Checkbox } from "@poketto/ui";
import { Label } from "@poketto/ui";
import { useState } from "react";

export default function IndeterminateCheckbox() {
  const [items, setItems] = useState([false, false, false]);
  
  const allChecked = items.every(Boolean);
  const someChecked = items.some(Boolean) && !allChecked;

  return (
    <div className="space-y-4">
      <div className="flex items-center space-x-2">
        <Checkbox 
          id="all"
          checked={allChecked}
          // @ts-ignore
          indeterminate={someChecked}
          onCheckedChange={(checked) => {
            setItems([checked as boolean, checked as boolean, checked as boolean]);
          }}
        />
        <Label htmlFor="all" className="font-semibold">Select all</Label>
      </div>
      <div className="ml-6 space-y-2">
        {items.map((checked, index) => (
          <div key={index} className="flex items-center space-x-2">
            <Checkbox
              id={`item-${index}`}
              checked={checked}
              onCheckedChange={(value) => {
                const newItems = [...items];
                newItems[index] = value as boolean;
                setItems(newItems);
              }}
            />
            <Label htmlFor={`item-${index}`}>Item {index + 1}</Label>
          </div>
        ))}
      </div>
    </div>
  );
}
```

## Props

<ParamField path="checked" type="boolean | 'indeterminate'">
  The controlled checked state of the checkbox.
</ParamField>

<ParamField path="defaultChecked" type="boolean">
  The default checked state for uncontrolled checkboxes.
</ParamField>

<ParamField path="onCheckedChange" type="(checked: boolean | 'indeterminate') => void">
  Event handler called when the checked state changes.
</ParamField>

<ParamField path="disabled" type="boolean">
  When `true`, disables the checkbox and reduces opacity.
</ParamField>

<ParamField path="required" type="boolean">
  When `true`, indicates that the checkbox must be checked.
</ParamField>

<ParamField path="name" type="string">
  The name of the checkbox (useful for form submission).
</ParamField>

<ParamField path="value" type="string">
  The value of the checkbox (useful for form submission).
</ParamField>

<ParamField path="className" type="string">
  Additional CSS classes to apply to the checkbox.
</ParamField>

## Form Integration

The Checkbox component integrates seamlessly with React Hook Form:

```tsx
import { Checkbox } from "@poketto/ui";
import { Button } from "@poketto/ui";
import { useForm } from "react-hook-form";
import {
  Form,
  FormControl,
  FormDescription,
  FormField,
  FormItem,
  FormLabel,
  FormMessage,
} from "@poketto/ui";

interface FormData {
  marketing: boolean;
  newsletter: boolean;
  terms: boolean;
}

export default function CheckboxForm() {
  const form = useForm<FormData>({
    defaultValues: {
      marketing: false,
      newsletter: false,
      terms: false,
    },
  });

  const onSubmit = (data: FormData) => {
    console.log(data);
  };

  return (
    <Form {...form}>
      <form onSubmit={form.handleSubmit(onSubmit)} className="space-y-6">
        <FormField
          control={form.control}
          name="marketing"
          render={({ field }) => (
            <FormItem className="flex flex-row items-start space-x-3 space-y-0">
              <FormControl>
                <Checkbox
                  checked={field.value}
                  onCheckedChange={field.onChange}
                />
              </FormControl>
              <div className="space-y-1 leading-none">
                <FormLabel>Marketing emails</FormLabel>
                <FormDescription>
                  Receive emails about new products and features.
                </FormDescription>
              </div>
            </FormItem>
          )}
        />
        <FormField
          control={form.control}
          name="newsletter"
          render={({ field }) => (
            <FormItem className="flex flex-row items-start space-x-3 space-y-0">
              <FormControl>
                <Checkbox
                  checked={field.value}
                  onCheckedChange={field.onChange}
                />
              </FormControl>
              <div className="space-y-1 leading-none">
                <FormLabel>Newsletter</FormLabel>
                <FormDescription>
                  Subscribe to our monthly newsletter.
                </FormDescription>
              </div>
            </FormItem>
          )}
        />
        <FormField
          control={form.control}
          name="terms"
          render={({ field }) => (
            <FormItem className="flex flex-row items-start space-x-3 space-y-0">
              <FormControl>
                <Checkbox
                  checked={field.value}
                  onCheckedChange={field.onChange}
                />
              </FormControl>
              <div className="space-y-1 leading-none">
                <FormLabel>Accept terms and conditions</FormLabel>
                <FormMessage />
              </div>
            </FormItem>
          )}
        />
        <Button type="submit">Submit</Button>
      </form>
    </Form>
  );
}
```
