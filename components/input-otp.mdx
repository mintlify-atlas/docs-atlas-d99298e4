title: Input OTP

The Input OTP component provides an accessible one-time password input, perfect for authentication flows.

## Installation

```tsx
import { InputOTP, InputOTPGroup, InputOTPSlot } from "@poketto/ui/input-otp"
```

## Usage

### Basic OTP input

```tsx
import { InputOTP, InputOTPGroup, InputOTPSlot } from "@poketto/ui/input-otp"

export default function Example() {
  return (
    <InputOTP maxLength={6}>
      <InputOTPGroup>
        <InputOTPSlot index={0} />
        <InputOTPSlot index={1} />
        <InputOTPSlot index={2} />
        <InputOTPSlot index={3} />
        <InputOTPSlot index={4} />
        <InputOTPSlot index={5} />
      </InputOTPGroup>
    </InputOTP>
  )
}
```

### With separator

```tsx
<InputOTP maxLength={6}>
  <InputOTPGroup>
    <InputOTPSlot index={0} />
    <InputOTPSlot index={1} />
    <InputOTPSlot index={2} />
  </InputOTPGroup>
  <InputOTPSeparator />
  <InputOTPGroup>
    <InputOTPSlot index={3} />
    <InputOTPSlot index={4} />
    <InputOTPSlot index={5} />
  </InputOTPGroup>
</InputOTP>
```

### Controlled

```tsx
import { useState } from "react"

export default function Example() {
  const [value, setValue] = useState("")

  return (
    <div className="space-y-2">
      <InputOTP
        maxLength={6}
        value={value}
        onChange={setValue}
      >
        <InputOTPGroup>
          <InputOTPSlot index={0} />
          <InputOTPSlot index={1} />
          <InputOTPSlot index={2} />
          <InputOTPSlot index={3} />
          <InputOTPSlot index={4} />
          <InputOTPSlot index={5} />
        </InputOTPGroup>
      </InputOTP>
      <p className="text-sm text-muted-foreground">
        Entered: {value || "None"}
      </p>
    </div>
  )
}
```

## Props

<ParamField path="maxLength" type="number" required>
  Maximum number of characters
</ParamField>

<ParamField path="value" type="string">
  The controlled value
</ParamField>

<ParamField path="onChange" type="(value: string) => void">
  Event handler for value changes
</ParamField>

<ParamField path="disabled" type="boolean">
  Whether the input is disabled
</ParamField>

## Accessibility

- Automatic focus management
- Paste support for quick entry
- Keyboard navigation between slots
- Screen reader support
