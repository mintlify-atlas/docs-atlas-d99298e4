## Introduction

The RadioGroup component is built on top of Radix UI's RadioGroup primitive and provides a fully accessible way to select one option from a set of mutually exclusive options.

## Installation

The RadioGroup component is available as part of the `@poketto/ui` package:

```tsx
import { RadioGroup, RadioGroupItem } from "@poketto/ui";
```

## Basic Usage

```tsx
import { RadioGroup, RadioGroupItem } from "@poketto/ui";
import { Label } from "@poketto/ui";

export default function Example() {
  return (
    <RadioGroup defaultValue="option-1">
      <div className="flex items-center space-x-2">
        <RadioGroupItem value="option-1" id="option-1" />
        <Label htmlFor="option-1">Option 1</Label>
      </div>
      <div className="flex items-center space-x-2">
        <RadioGroupItem value="option-2" id="option-2" />
        <Label htmlFor="option-2">Option 2</Label>
      </div>
    </RadioGroup>
  );
}
```

### Controlled Radio Group

```tsx
import { RadioGroup, RadioGroupItem } from "@poketto/ui";
import { Label } from "@poketto/ui";
import { useState } from "react";

export default function ControlledRadio() {
  const [value, setValue] = useState("comfortable");

  return (
    <div className="space-y-4">
      <RadioGroup value={value} onValueChange={setValue}>
        <div className="flex items-center space-x-2">
          <RadioGroupItem value="comfortable" id="comfortable" />

        </div>
      </RadioGroup>

      </p>
    </div>
  );
}
```

## Props

### RadioGroup

<ParamField path="value" type="string">
  The controlled value of the radio group.
</ParamField>

<ParamField path="defaultValue" type="string">
  The default value for uncontrolled radio groups.
</ParamField>

<ParamField path="onValueChange" type="(value: string) => void">
  Event handler called when the value changes.
</ParamField>

<ParamField path="disabled" type="boolean">
  When `true`, disables the entire radio group.
</ParamField>

<ParamField path="required" type="boolean">
  When `true`, indicates that a selection is required.
</ParamField>

<ParamField path="name" type="string">
  The name of the radio group (useful for form submission).
</ParamField>

<ParamField path="className" type="string">
  Additional CSS classes to apply to the radio group container.
</ParamField>

### RadioGroupItem

<ParamField path="value" type="string" required>
  The value of the radio item.
</ParamField>

<ParamField path="disabled" type="boolean">
  When `true`, disables the radio item.
</ParamField>

<ParamField path="id" type="string">
  The id attribute for the radio item (used with Label).
</ParamField>

<ParamField path="className" type="string">
  Additional CSS classes to apply to the radio item.
</ParamField>

## Form Integration

The RadioGroup component integrates seamlessly with React Hook Form:

```tsx
import { RadioGroup, RadioGroupItem } from "@poketto/ui";
import { Button } from "@poketto/ui";
import { useForm } from "react-hook-form";
import {
  Form,
  FormControl,
  FormDescription,
  FormField,
  FormItem,
  FormLabel,
  FormMessage,
} from "@poketto/ui";
import { Label } from "@poketto/ui";

interface FormData {
  notificationType: string;
}

export default function RadioForm() {
  const form = useForm<FormData>({
    defaultValues: {
      notificationType: "email",
    },
  });

  const onSubmit = (data: FormData) => {
    console.log(data);
  };

  return (
    <Form {...form}>
      <form onSubmit={form.handleSubmit(onSubmit)} className="space-y-6">
        <FormField
          control={form.control}
          name="notificationType"
          render={({ field }) => (
            <FormItem className="space-y-3">
              <FormLabel>Notification method</FormLabel>
              <FormControl>
                <RadioGroup
                  onValueChange={field.onChange}
                  defaultValue={field.value}
                  className="flex flex-col space-y-1"
                >
                  <FormItem className="flex items-center space-x-3 space-y-0">
                    <FormControl>
                      <RadioGroupItem value="email" />
                    </FormControl>
                    <FormLabel className="font-normal">
                      Email
                    </FormLabel>
                  </FormItem>
                  <FormItem className="flex items-center space-x-3 space-y-0">
                    <FormControl>
                      <RadioGroupItem value="sms" />
                    </FormControl>
                    <FormLabel className="font-normal">
                      SMS
                    </FormLabel>
                  </FormItem>
                  <FormItem className="flex items-center space-x-3 space-y-0">
                    <FormControl>
                      <RadioGroupItem value="push" />
                    </FormControl>
                    <FormLabel className="font-normal">
                      Push notification
                    </FormLabel>
                  </FormItem>
                </RadioGroup>
              </FormControl>
              <FormDescription>
                Choose how you want to receive notifications.
              </FormDescription>
              <FormMessage />
            </FormItem>
          )}
        />
        <Button type="submit">Submit</Button>
      </form>
    </Form>
  );
}
```
