## Introduction

The Textarea component is a styled multi-line text input field that extends the native HTML textarea element with consistent styling and accessibility features.

## Installation

The Textarea component is available as part of the `@poketto/ui` package:

```tsx
import { Textarea } from "@poketto/ui";
```

## Basic Usage

```tsx
import { Textarea } from "@poketto/ui";

export default function Example() {
  return <Textarea placeholder="Type your message here" />;
}
```

## Examples

### With Label

```tsx
import { Textarea } from "@poketto/ui";
import { Label } from "@poketto/ui";

export default function TextareaWithLabel() {
  return (
    <div className="space-y-2">
      <Label htmlFor="message">Message</Label>
      <Textarea id="message" placeholder="Enter your message" />
    </div>
  );
}
```

### With Character Count

```tsx
import { Textarea } from "@poketto/ui";
import { Label } from "@poketto/ui";
import { useState } from "react";

export default function TextareaWithCount() {
  const [value, setValue] = useState("");
  const maxLength = 200;

  return (
    <div className="space-y-2">
      <Label htmlFor="bio">Bio</Label>
      <Textarea
        id="bio"
        placeholder="Tell us about yourself"
        value={value}
        onChange={(e) => setValue(e.target.value)}
        maxLength={maxLength}
      />
      <p className="text-sm text-muted-foreground text-right">
        {value.length}/{maxLength} characters
      </p>
    </div>
  );
}
```

### Controlled Textarea

```tsx
import { Textarea } from "@poketto/ui";
import { useState } from "react";

export default function ControlledTextarea() {
  const [value, setValue] = useState("");

  return (
    <div className="space-y-2">
      <Textarea
        value={value}
        onChange={(e) => setValue(e.target.value)}
        placeholder="Type something..."
      />
      <p className="text-sm text-muted-foreground">
        Word count: {value.split(/\s+/).filter(Boolean).length}
      </p>
    </div>
  );
}
```

### With Error State

```tsx
import { Textarea } from "@poketto/ui";
import { Label } from "@poketto/ui";
import { useState } from "react";

export default function TextareaWithError() {
  const [value, setValue] = useState("");
  const minLength = 10;
  const hasError = value.length > 0 && value.length < minLength;

  return (
    <div className="space-y-2">
      <Label htmlFor="feedback">Feedback</Label>
      <Textarea
        id="feedback"
        value={value}
        onChange={(e) => setValue(e.target.value)}
        placeholder="Share your feedback"
        className={hasError ? "border-destructive" : ""}
      />
      {hasError && (
        <p className="text-sm text-destructive">
          Feedback must be at least {minLength} characters.
        </p>
      )}
    </div>
  );
}
```

## Props

<ParamField path="placeholder" type="string">
  Placeholder text displayed when the textarea is empty.
</ParamField>

<ParamField path="disabled" type="boolean">
  When `true`, disables the textarea and reduces opacity.
</ParamField>

<ParamField path="readOnly" type="boolean">
  When `true`, makes the textarea read-only.
</ParamField>

<ParamField path="value" type="string">
  The controlled value of the textarea.
</ParamField>

<ParamField path="defaultValue" type="string">
  The default value for uncontrolled textareas.
</ParamField>

<ParamField path="onChange" type="(event: React.ChangeEvent<HTMLTextAreaElement>) => void">
  Change event handler called when the textarea value changes.
</ParamField>

<ParamField path="rows" type="number">
  The number of visible text lines for the textarea.
</ParamField>

<ParamField path="maxLength" type="number">
  The maximum number of characters allowed.
</ParamField>

<ParamField path="required" type="boolean">
  When `true`, indicates that the textarea must be filled.
</ParamField>

<ParamField path="className" type="string">
  Additional CSS classes to apply to the textarea.
</ParamField>

The Textarea component accepts all standard HTML textarea attributes (`name`, `id`, `cols`, etc.).

## Form Integration

The Textarea component integrates seamlessly with React Hook Form:

```tsx
import { Textarea } from "@poketto/ui";
import { Button } from "@poketto/ui";
import { useForm } from "react-hook-form";
import {
  Form,
  FormControl,
  FormDescription,
  FormField,
  FormItem,
  FormLabel,
  FormMessage,
} from "@poketto/ui";

interface FormData {
  bio: string;
  feedback: string;
}

export default function TextareaForm() {
  const form = useForm<FormData>({
    defaultValues: {
      bio: "",
      feedback: "",
    },
  });

  const onSubmit = (data: FormData) => {
    console.log(data);
  };

  return (
    <Form {...form}>
      <form onSubmit={form.handleSubmit(onSubmit)} className="space-y-6">
        <FormField
          control={form.control}
          name="bio"
          render={({ field }) => (
            <FormItem>
              <FormLabel>Bio</FormLabel>
              <FormControl>
                <Textarea
                  placeholder="Tell us a little bit about yourself"
                  className="resize-none"
                  {...field}
                />
              </FormControl>
              <FormDescription>
                This will be displayed on your profile.
              </FormDescription>
              <FormMessage />
            </FormItem>
          )}
        />
        <FormField
          control={form.control}
          name="feedback"
          render={({ field }) => (
            <FormItem>
              <FormLabel>Feedback</FormLabel>
              <FormControl>
                <Textarea
                  placeholder="Share your thoughts"
                  className="min-h-[120px]"
                  {...field}
                />
              </FormControl>
              <FormMessage />
            </FormItem>
          )}
        />
        <Button type="submit">Submit</Button>
      </form>
    </Form>
  );
}
```
