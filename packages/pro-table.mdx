## Overview

`@poketto/pro-table` provides powerful, feature-rich table components for React applications. Built on [TanStack Table v8](https://tanstack.com/table/v8), it offers two main components:

- **DataTable** - Low-level table primitives for full customization
- **ProTable** - High-level table with built-in features (search, filters, pagination)

## Features

- ✅ Server-side pagination, sorting, and filtering
- 🔍 Built-in search functionality
- 📌 Column pinning (left/right)
- 👁️ Column visibility toggles
- ⚡ Optimized for large datasets
- 🎨 Fully customizable styling
- 📱 Responsive design
- ♿ Accessible by default

## Quick Start

### Basic ProTable Example

```tsx
import { ProTable } from '@poketto/pro-table/pro-table'
import type { ColumnDef } from '@poketto/pro-table/pro-table'

interface User {
  id: string
  name: string
  email: string
  role: string
}

const columns: ColumnDef<User>[] = [
  {
    accessorKey: 'name',
    header: 'Name',
  },
  {
    accessorKey: 'email',
    header: 'Email',
  },
  {
    accessorKey: 'role',
    header: 'Role',
  },
]

export function UsersTable() {
  const [data, setData] = useState<User[]>([])
  const [isLoading, setIsLoading] = useState(false)
  const [pagination, setPagination] = useState({
    pageIndex: 0,
    pageSize: 10,
    total: 0,
  })

  const fetchData = async () => {
    setIsLoading(true)
    const response = await fetch('/api/users')
    const result = await response.json()
    setData(result.data)
    setPagination(prev => ({ ...prev, total: result.total }))
    setIsLoading(false)
  }

  useEffect(() => {
    fetchData()
  }, [pagination.pageIndex, pagination.pageSize])

  return (
    <ProTable
      columns={columns}
      data={data}
      isLoading={isLoading}
      pagination={{
        ...pagination,
        onPageChange: (pageIndex) => 
          setPagination(prev => ({ ...prev, pageIndex })),
        onPageSizeChange: (pageSize) => 
          setPagination(prev => ({ ...prev, pageSize, pageIndex: 0 })),
      }}
      onRefresh={fetchData}
    />
  )
}
```

## ProTable Component

### Props

<ResponseField name="columns" type="ColumnDef<TData>[]" required>
  Array of column definitions
</ResponseField>

<ResponseField name="data" type="TData[]" required>
  Array of data items to display
</ResponseField>

<ResponseField name="isLoading" type="boolean" required>
  Loading state of the table
</ResponseField>

<ResponseField name="pagination" type="PaginationProps">
  Pagination configuration:
  - `pageIndex: number` - Current page (0-indexed)
  - `pageSize: number` - Items per page
  - `total: number` - Total number of items
  - `onPageChange: (page: number) => void`
  - `onPageSizeChange: (size: number) => void`
  - `quickJump?: boolean` - Enable page jump input
</ResponseField>

<ResponseField name="toolbar" type="React.ReactNode">
  Custom toolbar component rendered above the table
</ResponseField>

<ResponseField name="initialState" type="TableOptions['initialState']">
  Initial table state (sorting, filters, etc.)
</ResponseField>

<ResponseField name="onRefresh" type="() => void">
  Callback when refresh button is clicked
</ResponseField>

<ResponseField name="onSearch" type="(params: SearchParams) => void">
  Callback when search is performed
</ResponseField>

<ResponseField name="enableSearch" type="boolean" default="true">
  Enable/disable search functionality
</ResponseField>

### Server-Side Operations

```tsx
function UsersTable() {
  const [data, setData] = useState<User[]>([])
  const [isLoading, setIsLoading] = useState(false)
  const [pagination, setPagination] = useState({
    pageIndex: 0,
    pageSize: 10,
    total: 0,
  })
  const [search, setSearch] = useState<SearchParams>({})

  const fetchData = async () => {
    setIsLoading(true)
    try {
      const params = new URLSearchParams({
        page: String(pagination.pageIndex),
        pageSize: String(pagination.pageSize),
        ...search,
      })
      const response = await fetch(`/api/users?${params}`)
      const result = await response.json()
      setData(result.data)
      setPagination(prev => ({ ...prev, total: result.total }))
    } finally {
      setIsLoading(false)
    }
  }

  useEffect(() => {
    fetchData()
  }, [pagination.pageIndex, pagination.pageSize, search])

  return (
    <ProTable
      columns={columns}
      data={data}
      isLoading={isLoading}
      pagination={{
        ...pagination,
        onPageChange: (pageIndex) => 
          setPagination(prev => ({ ...prev, pageIndex })),
        onPageSizeChange: (pageSize) => 
          setPagination(prev => ({ ...prev, pageSize, pageIndex: 0 })),
      }}
      onSearch={(params) => setSearch(params)}
      onRefresh={fetchData}
    />
  )
}
```

## DataTable Component

For more control, use the low-level `DataTable` components:

```tsx
import {
  DataTable,
  DataTablePagination,
  DataTableToolbar,
  DataTableSearch,
} from '@poketto/pro-table/data-table'
import { useReactTable } from '@tanstack/react-table'

function CustomTable() {
  const table = useReactTable({
    data,
    columns,
    // ... your table config
  })

  return (
    <div className="space-y-4">
      <DataTableSearch table={table} />
      <DataTableToolbar table={table} />
      <div className="rounded-md border">
        <DataTable table={table} isLoading={isLoading} />
        <DataTablePagination table={table} pagination={pagination} />
      </div>
    </div>
  )
}
```

## Performance Tips

1. **Memoize columns** to prevent unnecessary re-renders:

```tsx
const columns = useMemo<ColumnDef<User>[]>(
  () => [
    // column definitions
  ],
  []
)
```

2. **Use server-side pagination** for large datasets
3. **Enable column virtualization** for tables with many columns
4. **Debounce search inputs** to reduce API calls
