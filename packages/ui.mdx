## Overview

`@poketto/ui` is a complete UI component library for React applications, built on top of [shadcn/ui](https://ui.shadcn.com/). It provides 40+ carefully crafted components that follow modern design principles and accessibility standards.

All components are:
- **Fully typed** with TypeScript
- **Accessible** following WAI-ARIA guidelines  
- **Customizable** via Tailwind CSS and CVA variants
- **Tree-shakeable** with granular imports
- **Framework-agnostic** (works with Next.js, Vite, Remix, etc.)

## Installation

```bash
pnpm add @poketto/ui
```

## Quick Start

### Setup Tailwind Configuration

Import the base Tailwind config in your project:

```js title="tailwind.config.js"
import uiConfig from '@poketto/ui/tailwind.config.ts'

export default {
  ...uiConfig,
  content: [
    './src/**/*.{ts,tsx}',
    './node_modules/@poketto/ui/src/**/*.{ts,tsx}',
  ],
}
```

### Import Global Styles

Add the global CSS file to your app entry point:

```tsx title="app/layout.tsx"
import '@poketto/ui/globals.css'
```

## Component Imports

Components can be imported individually for optimal tree-shaking:

```tsx
import { Button } from '@poketto/ui/button'
import { Card, CardHeader, CardContent } from '@poketto/ui/card'
import { Input } from '@poketto/ui/input'
import { Label } from '@poketto/ui/label'

export function LoginForm() {
  return (
    <Card>
      <CardHeader>
        <h2>Sign In</h2>
      </CardHeader>
      <CardContent className="space-y-4">
        <div>
          <Label htmlFor="email">Email</Label>
          <Input id="email" type="email" />
        </div>
        <div>
          <Label htmlFor="password">Password</Label>
          <Input id="password" type="password" />
        </div>
        <Button className="w-full">Sign In</Button>
      </CardContent>
    </Card>
  )
}
```

## Customization

### Component Variants

Components use `class-variance-authority` for type-safe variants:

```tsx
import { Button } from '@poketto/ui/button'

<Button variant="default">Default</Button>
<Button variant="destructive">Delete</Button>
<Button variant="outline">Outline</Button>
<Button variant="secondary">Secondary</Button>
<Button variant="ghost">Ghost</Button>
<Button variant="link">Link</Button>

<Button size="default">Default</Button>
<Button size="sm">Small</Button>
<Button size="lg">Large</Button>
<Button size="icon">🔍</Button>
```

### Theming

Customize the design system via CSS variables:

```css title="globals.css"
:root {
  --background: 0 0% 100%;
  --foreground: 222.2 84% 4.9%;
  --primary: 221.2 83.2% 53.3%;
  --primary-foreground: 210 40% 98%;
  /* ... */
}

.dark {
  --background: 222.2 84% 4.9%;
  --foreground: 210 40% 98%;
  /* ... */
}
```

### Extending Components

All components accept `className` for customization:

```tsx
import { Button } from '@poketto/ui/button'
import { cn } from '@poketto/ui/utils'

export function CustomButton({ className, ...props }) {
  return (
    <Button 
      className={cn('rounded-full px-8', className)}
      {...props}
    />
  )
}
```

## Hooks

### useToast

Display toast notifications:

```tsx
import { useToast } from '@poketto/ui/hooks/use-toast'

export function SaveButton() {
  const { toast } = useToast()

  const handleSave = () => {
    toast({
      title: 'Saved successfully',
      description: 'Your changes have been saved.',
    })
  }

  return <Button onClick={handleSave}>Save</Button>
}
```

### useMobile

Detect mobile viewport:

```tsx
import { useMobile } from '@poketto/ui/hooks/use-mobile'

export function ResponsiveComponent() {
  const isMobile = useMobile()

  return isMobile ? <MobileView /> : <DesktopView />
}
```

## Examples

### Form with Validation

```tsx
import { zodResolver } from '@hookform/resolvers/zod'
import { useForm } from 'react-hook-form'
import * as z from 'zod'
import { Button } from '@poketto/ui/button'
import {
  Form,
  FormControl,
  FormField,
  FormItem,
  FormLabel,
  FormMessage,
} from '@poketto/ui/form'
import { Input } from '@poketto/ui/input'

const formSchema = z.object({
  username: z.string().min(2).max(50),
  email: z.string().email(),
})

export function ProfileForm() {
  const form = useForm<z.infer<typeof formSchema>>({
    resolver: zodResolver(formSchema),
  })

  function onSubmit(values: z.infer<typeof formSchema>) {
    console.log(values)
  }

  return (
    <Form {...form}>
      <form onSubmit={form.handleSubmit(onSubmit)} className="space-y-8">
        <FormField
          control={form.control}
          name="username"
          render={({ field }) => (
            <FormItem>
              <FormLabel>Username</FormLabel>
              <FormControl>
                <Input placeholder="johndoe" {...field} />
              </FormControl>
              <FormMessage />
            </FormItem>
          )}
        />
        <Button type="submit">Submit</Button>
      </form>
    </Form>
  )
}
```

### Dialog Example

```tsx
import {
  Dialog,
  DialogContent,
  DialogDescription,
  DialogHeader,
  DialogTitle,
  DialogTrigger,
} from '@poketto/ui/dialog'
import { Button } from '@poketto/ui/button'

export function DeleteDialog() {
  return (
    <Dialog>
      <DialogTrigger asChild>
        <Button variant="destructive">Delete Account</Button>
      </DialogTrigger>
      <DialogContent>
        <DialogHeader>
          <DialogTitle>Are you absolutely sure?</DialogTitle>
          <DialogDescription>
            This action cannot be undone. This will permanently delete your
            account and remove your data from our servers.
          </DialogDescription>
        </DialogHeader>
      </DialogContent>
    </Dialog>
  )
}
```
